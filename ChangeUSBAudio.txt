STRING $url = "https://github.com/rog1n/flipperChangeUsbAudio/raw/main/zeDaManga.wav"; $destino = (Get-Item .).FullName + '\zeDaManga.wav'; Invoke-WebRequest -Uri $url -OutFile $destino; reg add "HKEY_CURRENT_USER\AppEvents\Schemes\Apps\.Default\DeviceConnect\.Current" /t REG_SZ /d $destino /f ; reg add "HKEY_CURRENT_USER\AppEvents\Schemes\Apps\.Default\DeviceDisconnect\.Current" /t REG_SZ /d $destino /f ;Exit;
